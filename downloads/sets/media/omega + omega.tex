\documentclass[tikz, preview]{standalone}
\usepackage[cm]{sfmath}
\begin{document}
\begin{tikzpicture}[
	every node/.append style={white},
	font=\sfseries,
	xscale=3, y scale=2,
	initial/.style={thick, white},
	successor/.style={shorten >=1.5mm, shorten <=1.5mm, white}
]
	\def\x{0}
	\def\w{1.75}
	\draw[initial] (\x, 0) -- (\x, 1);
	\draw[initial] (\x + \w, 0) -- (\x + \w, 1);
	\foreach \n in {2, ..., 30}{
		\pgfmathsetmacro\y{\x + pow(\n, -1.3)}
		\global\let\x\y
		\draw[successor] (\x, 0) -- (\x, 1);
		\draw[successor] (\x + \w, 0) -- (\x + \w, 1);
	}
	\def\x{-1}
	\foreach \n in {1, ..., 7}{
		\pgfmathsetmacro\y{\x + pow(\n, -1.3)}
		\global\let\x\y
		\path (\x, 0) node[below] {$\n$};
		\path (\x + \w, 0) node[below] {$\n'$};
	}
	\node[below=1mm] at (1.4, 0) {$\cdots$};
	\node[below=1mm] at (1.4+\w, 0) {$\cdots$};
\end{tikzpicture}
\end{document}
